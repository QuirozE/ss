% Overriding the creation of empty double pages
% By making an empty pages, all pending figures are printed
% Clears odd/even pair
% No headers on empty even pages
\makeatletter
\def \cleardoublepage{
    \clearpage % end current page and output pending figures
    \if@twoside
        \ifodd\c@page %c@page is the page counter
        \else % if even (ugly but the easiest with vanilla Tex)
            \mbox{} % some text is needed to get a new page?
            \thispagestyle{plain} % removes header
            \newpage
            %\if@twocolumn
            %    \hbox{}
            %    \newpage
            %\fi
        \fi
    \fi
}
\makeatother
